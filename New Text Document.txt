client.on('guildMemberAdd', member => {
	const channel = member.guild.channels.find(ch => ch.name === 'general');
	if (!channel) return;
	channel.send(`Welcome ${member}!`);
});

client.on('guildMemberRemove', member => {
	const channel = member.guild.channels.find(ch => ch.name === 'general');
	if (!channel) return;
	channel.send(`Bye ${member}.`);
});
